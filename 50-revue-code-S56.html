app.use('/ma-route', async (req: any, res: any) => {
  // Body parsing manuel pour POST/PUT
  const b: any = await new Promise<any>((resolve) => {
    if (req.body && Object.keys(req.body).length > 0) { resolve(req.body); return; }
    let data = '';
    req.on('data', (chunk: any) => { data += chunk; });
    req.on('end', () => { try { resolve(JSON.parse(data)); } catch { resolve({}); } });
  });
  const db = await getDb();           // ← getDb() obligatoire (pas createConnection)
  try {
    const [rows]: any = await db.execute('SELECT ...', [params]);
    res.json(rows);
  } catch(e: any) { res.status(500).json({ error: e.message }); }
  finally { await db.end(); }
});
